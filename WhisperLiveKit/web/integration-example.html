💻 集成代码示例

JavaScript SDK 方式（推荐）

<!-- 引入 SDK -->
<script src="http://your-server:8000/web/whisper-livekit-sdk.js"></script>

<script>
// 创建客户端
const client = new WhisperLiveKit({
    serverUrl: 'ws://your-server:8000',
    language: 'zh'
});

// 监听结果
client.onResult = (result) => {
    console.log('转录结果:', result.text);
};

// 开始录音
await client.start();

// 停止录音
client.stop();
</script>

WebSocket 原生方式

const ws = new WebSocket('ws://your-server:8000/asr');

ws.onopen = () => {
    console.log('已连接');
};

ws.onmessage = (event) => {
    const data = JSON.parse(event.data);
    if (data.buffer_transcription) {
        console.log('实时转录:', data.buffer_transcription);
    }
};
